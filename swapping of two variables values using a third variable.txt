#include <stdio.h>
#include <stdlib.h>

int main() {
    int a, b, c;

    printf("Enter two integers:\n");
    printf("a = ");
    scanf("%d", &a);
    printf("b = ");
    scanf("%d", &b);

    printf("\nBefore swapping:\n");
    printf("a = %d, b = %d\n", a, b);

    c = a;
    a = b;
    b = c;

    printf("\nAfter swapping using third variable:\n");
    printf("a = %d, b = %d\n", a, b);

    return 0;
}
