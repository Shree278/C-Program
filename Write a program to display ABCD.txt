#include <stdio.h>

int main() {
    int r, c, sp;

    for (r = 1; r <= 5; r++) {
        for (sp = 4; sp >= r; sp--) {
            printf(" ");
        }
        for (c = 1; c <= r; c++) {
            printf("%c", c + 64);
        }
        for (c = c - 2; c >= 1; c--) {
            printf("%c", c + 64);
        }
        printf("\n");
    }

    return 0;
}
