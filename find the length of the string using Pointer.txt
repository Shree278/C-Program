#include <stdio.h>

int main() {
    char str[100];
    char *ptr = str;

    printf("Enter string: ");
    fgets(str, sizeof(str), stdin);

    int length = 0;
    while (*ptr != '\0' && *ptr != '\n') {
        length++;
        ptr++;
    }

    printf("Length of the given string '%s': %d\n", str, length);

    return 0;
}
