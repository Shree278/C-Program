#include <stdio.h>

int main() {
    int a, b;

    printf("Enter two integers:\n");
    printf("a = ");
    scanf("%d", &a);
    printf("b = ");
    scanf("%d", &b);

    printf("\nBefore swapping:\n");
    printf("a = %d, b = %d\n", a, b);

    a = a + b;
    b = a - b;
    a = a - b;

    printf("\nAfter swapping without using third variable:\n");
    printf("a = %d, b = %d\n", a, b);

    return 0;
}
